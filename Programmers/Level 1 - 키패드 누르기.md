class Solution {
    public String solution(int[] numbers, String hand) {
        String answer = "";
        Coordinate pos = new Coordinate(hand);
        for (int num : numbers) {
            answer += pos.move(num);
        }
        return answer;
    }
    
    class Coordinate {
        private final int[][] keypad = new int[][] {
            {1,2,3},
            {4,5,6},
            {7,8,9},
            {-1,0,-2}
        };
        private int left = -1;
        private int right = -2;
        private String hand;
        
        public Coordinate (String hand) {
            this.hand = hand;
        }
        
        private int[] getPosition(int number) {
            int[] pos = new int[] { 0, 0 };

            for (int i = 0; i < keypad.length; i++)
            {
                for (int j = 0; j < keypad[i].length; j++)
                {
                    if (keypad[i][j] == number)
                    {
                        pos[0] = i;
                        pos[1] = j;
                    }
                }
            }
            
            return pos;
        }
        
        private String move (int num) {
            int [] pos, left, right;
            pos = getPosition(num);
            left = getPosition(this.left);
            right = getPosition(this.right);
            
            if (num == 1 || num == 4 || num == 7) { // 1,4,7ì¼ ê²½ìš° ì™¼ì†
