class Solution {

        int[] count = new int[number + 1];    
        for (int i = 1; i <= number; i++) {
            for (int j = 1; j <= number / i; j++) {
                count[i * j]++;
            }
        }

            if (count[i] > limit) {
                answer += power;
            } else {
                answer += count[i];
            }
        }

    }
}
